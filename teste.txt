PROGRAM exemplo;

VAR a, b: INTEGER;
/ teste comentairo /
BEGIN
/ teste comentairo /
  READ(a, b);
  IF a > b THEN
    BEGIN
      WRITE(a, 'maior do que', b)
    END;
  IF a == b THEN
    BEGIN
      WRITE(a, 'igual a', b)
    END;
  IF a < b THEN
    BEGIN
      WRITE(b, 'maior do que', a)
    END
END.
